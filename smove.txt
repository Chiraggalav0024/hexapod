#include <Wire.h>
#include <Adafruit_PWMServoDriver.h>
#include <Servo.h>

#define TOTAL_SERVOS 22
#define NUM_DIRECT_SERVOS 6

#define SERVOMIN 110
#define SERVOMAX 490

Adafruit_PWMServoDriver pca = Adafruit_PWMServoDriver(0x40);
Servo directServos[NUM_DIRECT_SERVOS];

int directPins[NUM_DIRECT_SERVOS] = {2,3,4,5,8,9};

int currentAngle[TOTAL_SERVOS];

// ================= POSES =================

struct Pose {
  const char* name;
  int angle[TOTAL_SERVOS];
};

Pose stand = {
"stand",
{90,96,90,118,96,90,119,80,90,114,94,90,77,120,90,90,129,90,49,98,56,90}
};

Pose stand0 = {
"stand0",
{124,79,110,74,125,107,63,131,101,63,135,90,143,53,90,173,47,81,74,93,96,90}
};

Pose stand1 = {
"stand1",
{20,158,110,169,39,98,160,40,102,167,51,90,40,154,93,61,146,81,74,93,96,90}
};

Pose* POSES[] = { &stand, &stand0, &stand1 };


// ================= BASIC FUNCTIONS =================

uint16_t angleToPulse(int angle){
  return map(angle,0,180,SERVOMIN,SERVOMAX);
}

void writeServo(int id,int angle){

  angle = constrain(angle,0,180);

  if(id < 16)
    pca.setPWM(id,0,angleToPulse(angle));
  else
    directServos[id-16].write(angle);
}

void moveToPose(Pose &p){

  bool moving = true;

  while(moving){

    moving = false;

    for(int i=0;i<TOTAL_SERVOS;i++){

      if(currentAngle[i] < p.angle[i]){
        currentAngle[i]++;
        moving = true;
      }
      else if(currentAngle[i] > p.angle[i]){
        currentAngle[i]--;
        moving = true;
      }

      writeServo(i,currentAngle[i]);
    }

    delay(8);
  }
}


// ================= SIMULTANEOUS MOVE =================

void moveSim(int ids[], int targets[], int count){

  bool moving = true;

  while(moving){

    moving = false;

    for(int i=0;i<count;i++){

      int id = ids[i];

      if(currentAngle[id] < targets[i]){
        currentAngle[id]++;
        moving = true;
      }
      else if(currentAngle[id] > targets[i]){
        currentAngle[id]--;
        moving = true;
      }

      writeServo(id,currentAngle[id]);
    }

    delay(8);
  }
}


// ================= SMOVE ENGINE =================

void doSMove(int poseIndex){

  Pose &homePose = *POSES[poseIndex];

  moveToPose(homePose);

  for(int cycle=0; cycle<3; cycle++){

    // Superphase A

    int s1[] = {1,7,13};
    int t1[] = {homePose.angle[1]-40,homePose.angle[7]+40,homePose.angle[13]-40};
    moveSim(s1,t1,3);

    int s2[] = {2,8,14};
    int t2[] = {homePose.angle[2]-30,homePose.angle[8]-30,homePose.angle[14]-30};
    moveSim(s2,t2,3);

    int s3[] = {1,7,13};
    int t3[] = {homePose.angle[1],homePose.angle[7],homePose.angle[13]};
    moveSim(s3,t3,3);

    int s4[] = {2,8,14,4,10,16};
    int t4[] = {homePose.angle[2],homePose.angle[8],homePose.angle[14],
                homePose.angle[4]+40,homePose.angle[10]+40,homePose.angle[16]-40};
    moveSim(s4,t4,6);


    // Superphase B

    int s5[] = {5,11,17};
    int t5[] = {homePose.angle[5]+30,homePose.angle[11]+30,homePose.angle[17]+30};
    moveSim(s5,t5,3);

    int s6[] = {4,10,16};
    int t6[] = {homePose.angle[4],homePose.angle[10],homePose.angle[16]};
    moveSim(s6,t6,3);

    int s7[] = {5,11,17,1,7,13};
    int t7[] = {homePose.angle[5],homePose.angle[11],homePose.angle[17],
                homePose.angle[1]-40,homePose.angle[7]+40,homePose.angle[13]-40};
    moveSim(s7,t7,6);
  }

  moveToPose(homePose);
}


// ================= SETUP =================

void setup(){

  Serial.begin(115200);

  pca.begin();
  pca.setPWMFreq(50);

  for(int i=0;i<NUM_DIRECT_SERVOS;i++)
    directServos[i].attach(directPins[i]);

  for(int i=0;i<TOTAL_SERVOS;i++){
    currentAngle[i] = 90;
    writeServo(i,90);
  }

  delay(1000);

  moveToPose(stand);

  Serial.println("SMove Test Ready");
  Serial.println("Commands: smove smove0 smove1");
}


// ================= LOOP =================

void loop(){

  if(Serial.available()){

    String cmd = Serial.readStringUntil('\n');
    cmd.trim();

    if(cmd == "smove"){
      doSMove(0);
      Serial.println("smove done");
    }

    if(cmd == "smove0"){
      doSMove(1);
      Serial.println("smove0 done");
    }

    if(cmd == "smove1"){
      doSMove(2);
      Serial.println("smove1 done");
    }
  }
}
