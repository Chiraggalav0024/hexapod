#include <Wire.h>
#include <Adafruit_PWMServoDriver.h>
#include <Servo.h>

#define TOTAL_SERVOS 22
#define NUM_DIRECT_SERVOS 6

#define SERVOMIN 110
#define SERVOMAX 490

Adafruit_PWMServoDriver pca = Adafruit_PWMServoDriver(0x40);

Servo directServos[NUM_DIRECT_SERVOS];
int directPins[NUM_DIRECT_SERVOS] = {2,3,4,5,8,9};

int currentAngle[TOTAL_SERVOS];

// -------- STAND NEUTRAL --------
int standneutral[TOTAL_SERVOS] = {
90,96,90,118,
96,90,119,80,
90,114,94,90,
77,120,90,90,
129,90,49,98,
56,90
};

// -------- SERVO WRITE --------
uint16_t angleToPulse(int angle){
  return map(angle,0,180,SERVOMIN,SERVOMAX);
}

void writeServo(int id,int angle){

  angle = constrain(angle,0,180);

  if(id < 16)
    pca.setPWM(id,0,angleToPulse(angle));
  else
    directServos[id-16].write(angle);
}

// -------- SIMULTANEOUS MOVE --------
void moveSim(int ids[], int targets[], int count){

  bool moving = true;

  while(moving){

    moving = false;

    for(int i=0;i<count;i++){

      int id = ids[i];

      if(currentAngle[id] < targets[i]){
        currentAngle[id]++;
        moving = true;
      }
      else if(currentAngle[id] > targets[i]){
        currentAngle[id]--;
        moving = true;
      }

      writeServo(id,currentAngle[id]);
    }

    delay(8);
  }
}

// -------- MOVE TO STAND --------
void moveToStand(){

  for(int i=0;i<TOTAL_SERVOS;i++){
    currentAngle[i] = standneutral[i];
    writeServo(i,currentAngle[i]);
  }
}

// -------- WALK CYCLE --------
void walkCycle(bool backward, bool finalCycle){

  int dir = backward ? -1 : 1;

  // PHASE 1 (lift tripod A)
  int s1[] = {1,7,13};
  int t1[] = {
    standneutral[1]-40,
    standneutral[7]+40,
    standneutral[13]-40
  };
  moveSim(s1,t1,3);
  delay(120);

  // PHASE 2 (swing A)
 int s2[] = {2,8,14};
int t2[] = {
  standneutral[2]-30*dir,
  standneutral[8]-30*dir,
  standneutral[14]+30*dir
};
  moveSim(s2,t2,3);
  delay(120);

  // PHASE 3 (place A)
  int s3[] = {1,7,13};
  int t3[] = {
    standneutral[1],
    standneutral[7],
    standneutral[13]
  };
  moveSim(s3,t3,3);
  delay(120);

  // PHASE 4
  int s4[] = {2,8,14,4,10,16};
  int t4[] = {
    standneutral[2],
    standneutral[8],
    standneutral[14],
    standneutral[4]+40,
    standneutral[10]+40,
    standneutral[16]-40
  };
  moveSim(s4,t4,6);
  delay(120);

  // PHASE 5
 int s5[] = {5,11,17};
int t5[] = {
  standneutral[5]-30*dir,
  standneutral[11]+30*dir,   // reversed
  standneutral[17]+30*dir
};
  moveSim(s5,t5,3);
  delay(120);

  // PHASE 6
  int s6[] = {4,10,16};
  int t6[] = {
    standneutral[4],
    standneutral[10],
    standneutral[16]
  };
  moveSim(s6,t6,3);
  delay(120);

  // PHASE 7
  if(!finalCycle){

    int s7[] = {5,11,17,1,7,13};

    int t7[] = {
      standneutral[5],
      standneutral[11],
      standneutral[17],

      standneutral[1]-40,
      standneutral[7]+40,
      standneutral[13]-40
    };

    moveSim(s7,t7,6);
  }
  else{

    int s7[] = {5,11,17};

    int t7[] = {
      standneutral[5],
      standneutral[11],
      standneutral[17]
    };

    moveSim(s7,t7,3);
  }

  delay(200);
}

// -------- FORWARD WALK --------
void forwardWalk(){

  walkCycle(false,false);
  walkCycle(false,false);
  walkCycle(false,true);
}

// -------- BACKWARD WALK --------
void backwardWalk(){

  walkCycle(true,false);
  walkCycle(true,false);
  walkCycle(true,true);
}

// -------- SETUP --------
void setup(){

  Serial.begin(115200);

  pca.begin();
  pca.setPWMFreq(50);

  for(int i=0;i<NUM_DIRECT_SERVOS;i++)
    directServos[i].attach(directPins[i]);

  delay(1000);

  moveToStand();

  Serial.println("Hexapod Ready");
  Serial.println("Type: fmove or bmove");
}

// -------- LOOP --------
void loop(){

  if(Serial.available()){

    String cmd = Serial.readStringUntil('\n');
    cmd.trim();

    if(cmd == "fmove"){
      Serial.println("Forward Move");
      forwardWalk();
    }

    else if(cmd == "bmove"){
      Serial.println("Backward Move");
      backwardWalk();
    }

  }
}
